        const canvas = document.getElementById('particleCanvas');
        const ctx = canvas.getContext('2d');
        let particles = [];

        function resizeCanvas() {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        }
        resizeCanvas();

        class Particle {
            constructor(x, y, color) {
                this.x = x; this.y = y;
                this.vx = (Math.random() - 0.5) * 12;
                this.vy = (Math.random() - 0.5) * 12;
                this.life = 1;
                this.decay = Math.random() * 0.02 + 0.01;
                this.size = Math.random() * 4 + 1;
                this.color = color;
            }
            update() {
                this.x += this.vx;
                this.y += this.vy;
                this.vy += 0.1;
                this.life -= this.decay;
                this.vx *= 0.98;
            }
            draw() {
                ctx.globalAlpha = this.life;
                ctx.fillStyle = this.color;
                ctx.shadowColor = this.color;
                ctx.shadowBlur = 8;
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
                ctx.fill();
                ctx.shadowBlur = 0;
            }
        }

        function spawnExplosion(x, y, colors, count = 60) {
            for (let i = 0; i < count; i++) {
                particles.push(new Particle(x, y, colors[Math.floor(Math.random() * colors.length)]));
            }
        }

        function animateParticles() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            particles = particles.filter(p => p.life > 0);
            particles.forEach(p => { p.update(); p.draw(); });
            ctx.globalAlpha = 1;
            requestAnimationFrame(animateParticles);
        }
        animateParticles();

        const realIndicators = ['according to','reported by','officials said','study finds','research shows','data suggests','sources confirm','analysis','published in','reuters','associated press','statement'];

        function analyzeNews() {
            const text = document.getElementById('newsInput').value.trim();
            if (!text) {
                showToast('Please enter a news headline to analyze');
                return;
            }

            const btn = document.getElementById('analyzeBtn');
            const btnText = document.getElementById('btnText');
            btn.classList.add('loading');
            btnText.innerHTML = '<span class="loading-spinner"></span>SCANNING...';

            // Trigger scan line
            const scanLine = document.getElementById('scanLine');
            scanLine.classList.remove('active');
            void scanLine.offsetWidth;
            scanLine.classList.add('active');

            setTimeout(() => {
                const lower = text.toLowerCase();
                let fakeScore = 0, realScore = 0;

                fakeIndicators.forEach(k => { if (lower.includes(k)) fakeScore += 2; });
                realIndicators.forEach(k => { if (lower.includes(k)) realScore += 2; });

                if (text === text.toUpperCase() && text.length > 10) fakeScore += 2;
                if ((text.match(/!/g) || []).length > 1) fakeScore += 1.5;
                if ((text.match(/\?!/g) || []).length > 0) fakeScore += 1;

                // Random factor for demo
                fakeScore += Math.random() * 1.5;
                realScore += Math.random() * 1.5;

                const isFake = fakeScore > realScore;
                const total = fakeScore + realScore || 1;
                const confidence = Math.min(95, Math.max(55, Math.round((Math.max(fakeScore, realScore) / total) * 100)));

                showResult(isFake, confidence);

                btn.classList.remove('loading');
                const currentBtnLabel = (window.elementSdk && window.elementSdk.config.button_text) || defaultConfig.button_text;
                btnText.innerHTML = '⚡ ' + currentBtnLabel;
            }, 2200);
        }
